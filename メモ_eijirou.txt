    int second_is_user = 0;



    Board board = create_board();
    print_board_for_debug(&board);

    //history = ; 盤面の履歴
    int winner = -1;

    for (int turn = 1; turn <= MAX_TURN; turn++){
        Action action;

        if (turn % 2){
            // 先手番
            if (first_is_user)

            else{
                action = get_ai_action(&board, history, turn);

        }

                display_action(action, turn);
            }
        }
